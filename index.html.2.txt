  <script>

    let products = [

      {
        name:"Demo Glasses 1",
        price:"2500",
        img1:"https://via.placeholder.com/300x280/5D4037/F5F5DC?text=Glasses+1A",
        img2:"https://via.placeholder.com/300x280/8D6E63/F5F5DC?text=Glasses+1B",
        type:"Glasses"
      },

      {
        name:"Demo Watch 1",
        price:"5500",
        img1:"https://via.placeholder.com/300x280/3E2723/F5F5DC?text=Watch+1A",
        img2:"https://via.placeholder.com/300x280/6D4C41/F5F5DC?text=Watch+1B",
        type:"Watch"
      }

    ];

    let isAdmin = false;

    function displayProducts(filter = 'All') {

      const container = document.getElementById('products');

      container.innerHTML = '';

      const filtered =
        filter === 'All'
        ? products
        : products.filter(p => p.type === filter);

      filtered.forEach((p,index) => {

        container.innerHTML += `

          <div class="product-card">

            <button class="delete-btn" onclick="deleteProduct(${index})">
              ×
            </button>

            <div class="img-slider">

              <img src="${p.img1}" alt="${p.name}" class="active">

              <img src="${p.img2}" alt="${p.name}">

              <div class="img-dots">
                <span class="dot active" onclick="changeImg(this,0)"></span>

                <span class="dot" onclick="changeImg(this,1)"></span>
              </div>

            </div>

            <h3>${p.name}</h3>

            <p class="product-type">${p.type}</p>

            <p class="price">
              Rs. ${p.price}
            </p>

            <p class="payment-info">
              ✅ COD | 📱 JazzCash Available
            </p>

            <button class="order-btn"
              onclick="orderNow('${p.name}','${p.price}')">

              Order on WhatsApp

            </button>

          </div>

        `;
      });

    }

    function filterProducts(type,event){

      displayProducts(type);

      document
        .querySelectorAll('.filters button')
        .forEach(btn => btn.classList.remove('active'));

      event.target.classList.add('active');
    }

    function changeImg(dot,index){

      const slider = dot.closest('.img-slider');

      slider
        .querySelectorAll('img')
        .forEach(img => img.classList.remove('active'));

      slider
        .querySelectorAll('.dot')
        .forEach(d => d.classList.remove('active'));

      slider.querySelectorAll('img')[index]
        .classList.add('active');

      dot.classList.add('active');
    }

    function toggleAdmin(){

      isAdmin = !isAdmin;

      document
        .getElementById('adminPanel')
        .classList.toggle('hidden');

      document.body.classList.toggle('admin-active');
    }

    function addProduct(){

      const name =
        document.getElementById('pName').value;

      const price =
        document.getElementById('pPrice').value;

      const img1 =
        document.getElementById('pImg1').value;

      const img2 =
        document.getElementById('pImg2').value;

      const type =
        document.getElementById('pType').value;

      if(name && price && img1 && img2){

        products.push({
          name,
          price,
          img1,
          img2,
          type
        });

        displayProducts();

        document.getElementById('pName').value = '';
        document.getElementById('pPrice').value = '';
        document.getElementById('pImg1').value = '';
        document.getElementById('pImg2').value = '';

        alert('Product added successfully!');

      } else {

        alert('Please fill all fields');

      }
    }

    function deleteProduct(index){

      if(confirm('Delete this product?')){

        products.splice(index,1);

        displayProducts();
      }
    }

    function saveToFile(){

      const dataStr =
        "data:text/json;charset=utf-8," +
        encodeURIComponent(
          JSON.stringify(products,null,2)
        );

      const dlAnchor =
        document.createElement('a');

      dlAnchor.setAttribute("href",dataStr);

      dlAnchor.setAttribute("download","products.json");

      dlAnchor.click();
    }

    function orderNow(name,price){

      const msg =
`Salam! HR Vision se order karna hai:

Product: ${name}
Price: Rs. ${price}

Payment: COD ya JazzCash

Please confirm`;

      window.open(
        `https://wa.me/923284854373?text=${encodeURIComponent(msg)}`,
        '_blank'
      );
    }

    displayProducts();
